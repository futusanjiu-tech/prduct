    <script type="module">
        import * as THREE from 'three';
        import { GLTFLoader } from 'three/addons/loaders/GLTFLoader.js';

        let scene, camera, renderer, model;
        let handDetected = false;
        let targetRotY = 0, currentRotY = 0;
        let targetScale = 1, currentScale = 1;
        let isDragging = false;
        let previousTouchX = 0;
        let initialPinchDist = 0;

        function initScene() {
            scene = new THREE.Scene();
            scene.background = new THREE.Color(0x050505);

            camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 0.1, 1000);
            camera.position.set(0, 0, 5);

            renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
            renderer.toneMapping = THREE.ACESFilmicToneMapping;
            renderer.toneMappingExposure = 1.4;
            document.getElementById('canvas-container').appendChild(renderer.domElement);

            scene.add(new THREE.AmbientLight(0xffffff, 1.2));
            const sun = new THREE.DirectionalLight(0xffffff, 2);
            sun.position.set(5, 10, 7.5);
            scene.add(sun);
            
            scene.add(new THREE.PointLight(0x00e5ff, 3, 10).set(-5, 0, 2));

            setupTouchEvents();
            
            // 部署逻辑：首先尝试自动加载
            tryAutoLoad();
        }

        function tryAutoLoad() {
            const status = document.getElementById('load-status');
            const manualBtn = document.getElementById('manual-btn');
            const ring = document.getElementById('ring');

            status.innerText = "Attempting Auto-Load...";
            
            const loader = new GLTFLoader();
            // 在服务器环境下，此处会自动成功
            loader.load('./20.glb', 
                (gltf) => handleModelReady(gltf),
                (xhr) => {
                    if(xhr.lengthComputable) {
                        status.innerText = `Loading: ${Math.round(xhr.loaded/xhr.total*100)}%`;
                    }
                },
                (err) => {
                    console.warn("Auto-load blocked or file missing. Switching to Manual Mode.");
                    status.innerText = "Security Sandbox Detected";
                    status.style.color = "#888";
                    ring.style.display = 'none';
                    manualBtn.style.display = 'block';
                }
            );
        }

        // 处理手动文件选择
        document.getElementById('file-input').onchange = (e) => {
            const file = e.target.files[0];
            if (!file) return;
            const url = URL.createObjectURL(file);
            const loader = new GLTFLoader();
            loader.load(url, (gltf) => handleModelReady(gltf));
        };

        function handleModelReady(gltf) {
            model = gltf.scene;
            const box = new THREE.Box3().setFromObject(model);
            const size = box.getSize(new THREE.Vector3());
            const center = box.getCenter(new THREE.Vector3());
            model.position.sub(center);
            const maxDim = Math.max(size.x, size.y, size.z);
            const baseScale = 2.5 / maxDim;
            model.scale.setScalar(baseScale);
            model.userData.baseScale = baseScale;
            scene.add(model);
            
            document.getElementById('loader-overlay').style.opacity = '0';
            setTimeout(() => {
                document.getElementById('loader-overlay').style.display = 'none';
                initHands();
            }, 800);
            animate();
        }

        function setupTouchEvents() {
            const container = document.getElementById('canvas-container');
            container.addEventListener('touchstart', (e) => {
                if (e.touches.length === 1) { isDragging = true; previousTouchX = e.touches[0].pageX; }
                else if (e.touches.length === 2) { initialPinchDist = Math.hypot(e.touches[0].pageX - e.touches[1].pageX, e.touches[0].pageY - e.touches[1].pageY); }
            }, { passive: false });
            container.addEventListener('touchmove', (e) => {
                if (isDragging && e.touches.length === 1) {
                    const deltaX = e.touches[0].pageX - previousTouchX;
                    targetRotY += deltaX * 0.01;
                    previousTouchX = e.touches[0].pageX;
                } else if (e.touches.length === 2) {
                    const currentDist = Math.hypot(e.touches[0].pageX - e.touches[1].pageX, e.touches[0].pageY - e.touches[1].pageY);
                    targetScale = Math.min(Math.max(targetScale * (currentDist / initialPinchDist), 0.3), 3.0);
                    initialPinchDist = currentDist;
                }
            }, { passive: false });
            container.addEventListener('touchend', () => { isDragging = false; });
        }

        function initHands() {
            const videoElement = document.getElementById('input-video');
            const hands = new Hands({ locateFile: (f) => `https://cdn.jsdelivr.net/npm/@mediapipe/hands/${f}` });
            hands.setOptions({ maxNumHands: 1, modelComplexity: 0, minDetectionConfidence: 0.6 });
            hands.onResults((results) => {
                if (results.multiHandLandmarks && results.multiHandLandmarks.length > 0) {
                    handDetected = true;
                    if (!isDragging) {
                        const landmarks = results.multiHandLandmarks[0];
                        targetRotY = (landmarks[0].x - 0.5) * Math.PI * 2;
                        const d = Math.hypot(landmarks[0].x - landmarks[9].x, landmarks[0].y - landmarks[9].y);
                        targetScale = Math.min(Math.max(d * 10, 0.4), 2.5);
                    }
                } else { handDetected = false; }
            });
            new Camera(videoElement, { onFrame: async () => { await hands.send({image: videoElement}); }, width: 480, height: 360 }).start();
        }

        function animate() {
            requestAnimationFrame(animate);
            if (model) {
                currentRotY = THREE.MathUtils.lerp(currentRotY, targetRotY, 0.1);
                currentScale = THREE.MathUtils.lerp(currentScale, targetScale, 0.1);
                model.rotation.y = currentRotY;
                model.scale.setScalar(currentScale * model.userData.baseScale);
                if (!handDetected && !isDragging) {
                    model.position.y = Math.sin(Date.now() * 0.002) * 0.05;
                    targetRotY += 0.003;
                }
            }
            renderer.render(scene, camera);
        }

        window.onload = initScene;
        window.onresize = () => {
            camera.aspect = window.innerWidth / window.innerHeight;
            camera.updateProjectionMatrix();
            renderer.setSize(window.innerWidth, window.innerHeight);
        };
    </script>
